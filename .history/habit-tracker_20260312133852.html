const modal = document.getElementById('habit-modal'); const openBtn = document.getElementById('add-habit-btn'); const cancelBtn = document.getElementById('modal-cancel'); const form = document.getElementById('modal-habit-form'); const habitsList = document.querySelector('.habits-list'); const emptyState = document.querySelector('.empty-state'); const pageStatus = document.getElementById('page-status'); const viewBtn = document.getElementById('view-habits-btn'); const STORAGE_KEY = 'mindtrace_habits_v1'; function showModal(){ modal.classList.add('show'); modal.setAttribute('aria-hidden','false'); document.getElementById('m-habit-name').focus(); } function hideModal(){ modal.classList.remove('show'); modal.setAttribute('aria-hidden','true'); form.reset(); } function getStoredHabits(){ try{ return JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]'); }catch(e){ return []; } } function saveStoredHabits(arr){ localStorage.setItem(STORAGE_KEY, JSON.stringify(arr)); } function addHabitToStorage(h){ const a = getStoredHabits(); a.unshift(h); saveStoredHabits(a); } // render stored habits into the in-page list (for preview) function renderHabitsPreview(){ const stored = getStoredHabits(); if(!habitsList) return; habitsList.innerHTML = ''; if(stored.length === 0){ if(emptyState) emptyState.style.display = ''; return; } stored.forEach(h => { const card = document.createElement('div'); card.className = 'habit-card'; card.innerHTML = `

${escapeHtml(h.name)}

${escapeHtml(h.frequency)}

🔥 ${h.streak||0}-day streak

`; habitsList.appendChild(card); }); if(emptyState) emptyState.style.display = 'none'; } openBtn && openBtn.addEventListener('click', function(){ showModal(); }); cancelBtn && cancelBtn.addEventListener('click', function(){ hideModal(); }); // navigate to a dedicated tracked-habits page instead of revealing the current list viewBtn && viewBtn.addEventListener('click', function(){ window.location.href = 'tracked-habits.html'; }); form && form.addEventListener('submit', function(e){ e.preventDefault(); const name = document.getElementById('m-habit-name').value.trim(); const frequency = document.getElementById('m-habit-frequency').value; const category = document.getElementById('m-habit-category').value; const description = document.getElementById('m-habit-description').value || ''; if(!name || !frequency) return; const habit = { name, frequency, category, description, created: Date.now(), streak: 0 }; addHabitToStorage(habit); // update preview immediately renderHabitsPreview(); showStatus('Habit saved'); hideModal(); }); // small helper to show brief status function showStatus(msg){ if(!pageStatus) return; pageStatus.hidden = false; pageStatus.querySelector('.status-text').textContent = msg; setTimeout(()=>{ pageStatus.hidden = true; }, 1800); } // simple escape to prevent accidental html injection function escapeHtml(s){ return String(s).replace(/&/g,'&').replace(//g,'>').replace(/"/g,'"'); } // delegate click for newly created check buttons document.addEventListener('click', function(e){ if(e.target && e.target.classList.contains('habit-checkbox')){ e.target.classList.toggle('done'); } }); // initialize preview from storage renderHabitsPreview();
